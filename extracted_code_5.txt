import json
from pprint import pprint

from recursive_improvement_engine import RecursiveImprovementEngine


def main():
    engine = RecursiveImprovementEngine(max_depth=4)
    engine.run(target_depth=4)

    # Summarize results
    summary = {
        "final_depth": engine.current_depth,
        "capability_history": engine.capability_history,
        "tools_created": list(engine.tool_registry.keys()),
    }

    print("\\n=== Experiment Summary ===")
    pprint(summary)

    # Persist to a JSON file for later analysis
    with open("depth_4_experiment_results.json", "w") as f:
        json.dump(summary, f, indent=2)


if __name__ == "__main__":
    main()
