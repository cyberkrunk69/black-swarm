def select_model(model_name: str) -> str:
    """
    Resolve a userâ€‘provided model identifier to a concrete model name.
    Raises:
        ValueError: If the model is not supported.
    """
    supported = {"gpt-4o", "gpt-4", "gpt-3.5-turbo"}
    if model_name in supported:
        return model_name
    raise ValueError(f"Unsupported model '{model_name}'. Supported models: {', '.join(supported)}")
